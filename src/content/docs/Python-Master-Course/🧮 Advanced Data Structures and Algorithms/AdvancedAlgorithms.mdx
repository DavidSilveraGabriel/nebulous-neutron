### Mergesort

```python
def mergesort(lista):

  medio = len(lista) // 2
  izquierda = lista[:medio]
  derecha = lista[medio:]
  izquierda = mergesort(izquierda)
  derecha = mergesort(derecha)
  return merge(izquierda, derecha)

def merge(izquierda, derecha):
   resultado = []
   i = j = 0
   while i < len(izquierda) and j < len(derecha):
      if izquierda[i] < derecha[j]:
          resultado.append(izquierda[i])
          i+=1
      else:
          resultado.append(derecha[j])
          j+=1
   resultado.extend(izquierda[i:])
   resultado.extend(derecha[j:])
   return resultado
```

### Quicksort
*  Algoritmo de ordenamiento por divide y vencer치s, recursivo.
* Elige un pivote, divide la lista en elementos menores y mayores al pivote, y los ordena de forma recursiva.
*   Tiempo de ejecuci칩n O(n log n) en el caso promedio, O(n^2) en el peor caso.

```python
def quicksort(lista):
  if len(lista) <= 1:
      return lista
  pivote = lista[len(lista) // 2]
  izquierda = [x for x in lista if x < pivote]
  medio = [x for x in lista if x == pivote]
  derecha = [x for x in lista if x > pivote]
  return quicksort(izquierda) + medio + quicksort(derecha)

lista = [5, 2, 8, 1, 9, 4]
lista_ordenada = quicksort(lista)
print(lista_ordenada) # [1, 2, 4, 5, 8, 9]
```

### Dijkstra
*  Algoritmo para encontrar el camino m치s corto desde un nodo inicial a todos los dem치s nodos de un grafo con pesos no negativos.

```python
import heapq

def dijkstra(grafo, inicio):
    distancias = {nodo: float('inf') for nodo in grafo}
    distancias[inicio] = 0
    cola_prioridad = [(0, inicio)]
    while cola_prioridad:
        distancia_actual, nodo_actual = heapq.heappop(cola_prioridad)
        if distancia_actual > distancias[nodo_actual]:
           continue
        for vecino, peso in grafo[nodo_actual].items():
           distancia = distancia_actual + peso
           if distancia < distancias[vecino]:
                distancias[vecino] = distancia
                heapq.heappush(cola_prioridad,(distancia, vecino))
    return distancias

grafo = {
    'A': {'B': 1, 'C': 4},
    'B': {'A': 1, 'D': 5, 'E': 2},
    'C': {'A': 4, 'F': 3},
    'D': {'B': 5},
    'E': {'B': 2, 'F': 1},
    'F': {'C': 3, 'E': 1}
}
distancias = dijkstra(grafo, 'A')
print(distancias) # {'A': 0, 'B': 1, 'C': 4, 'D': 6, 'E': 3, 'F': 4}
```

```python
    'E': {'B': 2, 'F': 1},
```

## 4. Algoritmos Greedy <Badge text="Soluciones Paso a Paso" variant="tip" />

Los algoritmos greedy (voraces) construyen soluciones paso a paso, eligiendo la opci칩n m치s prometedora en cada paso, con el objetivo de encontrar una soluci칩n 칩ptima global.

### Ejemplo: Problema de la Moneda con Algoritmo Greedy

```python

    monedas.sort(reverse=True)
    cambio = []
    for moneda in monedas:
        while cantidad >= moneda:
            cambio.append(moneda)
            cantidad -= moneda
    return cambio
```
